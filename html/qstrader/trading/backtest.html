380    def get_target_allocations(self):
381        """
382        Returns the target allocations as a Pandas DataFrame
383        utilising the same index as the equity curve with
384        forward-filled dates.
385
386        Returns
387        -------
388        `pd.DataFrame`
389            The datetime-indexed target allocations of the strategy.
390        """
391        equity_curve = self.get_equity_curve()
392        alloc_df = pd.DataFrame(self.target_allocations).set_index("Date")
393        alloc_df.index = alloc_df.index.date
394        alloc_df = alloc_df.reindex(index=equity_curve.index, method="ffill")
395        if self.burn_in_dt is not None:
396            alloc_df = alloc_df[self.burn_in_dt :]
397        return alloc_df

399    def run(self, results=False):
400        """
401        Execute the simulation engine by iterating over all
402        simulation events, rebalancing the quant trading
403        system at the appropriate schedule.
404
405        Parameters
406        ----------
407        results : `Boolean`, optional
408            Whether to output the current portfolio holdings
409        """
410        if settings.PRINT_EVENTS:
411            print("Beginning backtest simulation...")
412
413        stats = {"target_allocations": []}
414
415        for event in self.sim_engine:
416            # Output the system event and timestamp
417            dt = event.ts
418            if settings.PRINT_EVENTS:
419                print("(%s) - %s" % (event.ts, event.event_type))
420
421            # Update the simulated broker
422            self.broker.update(dt)
423
424            # Update any signals on a daily basis
425            if self.signals is not None and event.event_type == "market_close":
426                self.signals.update(dt)
427
428            # If we have hit a rebalance time then carry
429            # out a full run of the quant trading system
430            if self.burn_in_dt is not None:
431                if dt >= self.burn_in_dt:
432                    if self._is_rebalance_event(dt):
433                        if settings.PRINT_EVENTS:
434                            print("(%s) - trading logic " "and rebalance" % event.ts)
435                        self.qts(dt, stats=stats)
436            else:
437                if self._is_rebalance_event(dt):
438                    if settings.PRINT_EVENTS:
439                        print("(%s) - trading logic " "and rebalance" % event.ts)
440                    self.qts(dt, stats=stats)
441
442            # Out of market hours we want a daily
443            # performance update, but only if we
444            # are past the 'burn in' period
445            # if event.event_type == "market_close":
446            if self.burn_in_dt is not None:
447                if dt >= self.burn_in_dt:
448                    self._update_equity_curve(dt)
449            else:
450                self._update_equity_curve(dt)
451
452        self.target_allocations = stats["target_allocations"]
453
454        # At the end of the simulation output the
455        # portfolio holdings if desired
456        if results:
457            self.output_holdings()
458
459        if settings.PRINT_EVENTS:
460            print("Ending backtest simulation.")
