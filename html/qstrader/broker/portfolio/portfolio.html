 67    def _initialise_portfolio_with_cash(self):
 68        """
 69        Initialise the portfolio with a (default) currency Cash Asset
 70        with quantity equal to 'starting_cash'.
 71        """

 74        if self.starting_cash > 0.0:

 81        self.logger.info(
 82            '(%s) Funds subscribed to portfolio "%s" '

 85                self.portfolio_id,
 86                round(self.starting_cash, 2),
 87                round(self.starting_cash, 2)
 88            )

126    def subscribe_funds(self, dt, amount):
127        """
128        Credit funds to the portfolio.
129        """
130        if dt < self.current_dt:
131            raise ValueError(
132                'Subscription datetime (%s) is earlier than '
133                'current portfolio datetime (%s). Cannot '
134                'subscribe funds.' % (dt, self.current_dt)
135            )
136        self.current_dt = dt
137
138        if amount < 0.0:
139            raise ValueError(
140                'Cannot credit negative amount: '
141                '%s to the portfolio.' % amount
142            )
143
144        self.cash += amount
145
146        self.history.append(
147            PortfolioEvent.create_subscription(self.current_dt, amount, self.cash)
148        )
149
150        self.logger.info(
151            '(%s) Funds subscribed to portfolio "%s" '
152            '- Credit: %0.2f, Balance: %0.2f' % (
153                self.current_dt.strftime(settings.LOGGING["DATE_FORMAT"]),
154                self.portfolio_id, round(amount, 2),
155                round(self.cash, 2)
156            )
157        )

159    def withdraw_funds(self, dt, amount):
160        """
161        Withdraw funds from the portfolio if there is enough
162        cash to allow it.
163        """
164        # Check that amount is positive and that there is
165        # enough in the portfolio to withdraw the funds
166        if dt < self.current_dt:
167            raise ValueError(
168                'Withdrawal datetime (%s) is earlier than '
169                'current portfolio datetime (%s). Cannot '
170                'withdraw funds.' % (dt, self.current_dt)
171            )
172        self.current_dt = dt
173
174        if amount < 0:
175            raise ValueError(
176                'Cannot debit negative amount: '
177                '%0.2f from the portfolio.' % amount
178            )
179
180        if amount > self.cash:
181            raise ValueError(
182                'Not enough cash in the portfolio to '
183                'withdraw. %s withdrawal request exceeds '
184                'current portfolio cash balance of %s.' % (
185                    amount, self.cash
186                )
187            )
188
189        self.cash -= amount
190
191        self.history.append(
192            PortfolioEvent.create_withdrawal(self.current_dt, amount, self.cash)
193        )
194
195        self.logger.info(
196            '(%s) Funds withdrawn from portfolio "%s" '
197            '- Debit: %0.2f, Balance: %0.2f' % (
198                self.current_dt.strftime(settings.LOGGING["DATE_FORMAT"]),
199                self.portfolio_id, round(amount, 2),
200                round(self.cash, 2)
201            )
202        )

204    def transact_asset(self, txn):
205        """
206        Adjusts positions to account for a transaction.
207        """
208        if txn.dt < self.current_dt:
209            raise ValueError(
210                'Transaction datetime (%s) is earlier than '
211                'current portfolio datetime (%s). Cannot '
212                'transact assets.' % (txn.dt, self.current_dt)
213            )
214        self.current_dt = txn.dt
215
216        txn_share_cost = txn.price * txn.quantity
217        txn_total_cost = txn_share_cost + txn.commission
218
219        if txn_total_cost > self.cash:
220            if settings.PRINT_EVENTS:
221                print(
222                    'WARNING: Not enough cash in the portfolio to '
223                    'carry out transaction. Transaction cost of %s '
224                    'exceeds remaining cash of %s. Transaction '
225                    'will proceed with a negative cash balance.' % (
226                        txn_total_cost, self.cash
227                    )
228                )
229
230        self.pos_handler.transact_position(txn)
231
232        self.cash -= txn_total_cost
233
234        # Form Portfolio history details
235        direction = "LONG" if txn.direction > 0 else "SHORT"
236        description = "%s %s %s %0.2f %s" % (
237            direction, txn.quantity, txn.asset.upper(),
238            txn.price, datetime.datetime.strftime(txn.dt, "%d/%m/%Y")
239        )
240        if direction == "LONG":
241            pe = PortfolioEvent(
242                dt=txn.dt, type='asset_transaction',
243                description=description,
244                debit=round(txn_total_cost, 2), credit=0.0,
245                balance=round(self.cash, 2)
246            )
247            self.logger.info(
248                '(%s) Asset "%s" transacted LONG in portfolio "%s" '
249                '- Debit: %0.2f, Balance: %0.2f' % (
250                    txn.dt.strftime(settings.LOGGING["DATE_FORMAT"]),
251                    txn.asset, self.portfolio_id,
252                    round(txn_total_cost, 2), round(self.cash, 2)
253                )
254            )
255        else:
256            pe = PortfolioEvent(
257                dt=txn.dt, type='asset_transaction',
258                description=description,
259                debit=0.0, credit=-1.0 * round(txn_total_cost, 2),
260                balance=round(self.cash, 2)
261            )
262            self.logger.info(
263                '(%s) Asset "%s" transacted SHORT in portfolio "%s" '
264                '- Credit: %0.2f, Balance: %0.2f' % (
265                    txn.dt.strftime(settings.LOGGING["DATE_FORMAT"]),
266                    txn.asset, self.portfolio_id,
267                    -1.0 * round(txn_total_cost, 2), round(self.cash, 2)
268                )
269            )
270        self.history.append(pe)

272    def portfolio_to_dict(self):
273        """
274        Output the portfolio holdings information as a dictionary
275        with Assets as keys and sub-dictionaries as values.
276        This excludes cash.
277
278        Returns
279        -------
280        `dict`
281            The portfolio holdings.
282        """
283        holdings = {}
284        for asset, pos in self.pos_handler.positions.items():
285            holdings[asset] = {
286                "quantity": pos.net_quantity,
287                "market_value": pos.market_value,
288                "unrealised_pnl": pos.unrealised_pnl,
289                "realised_pnl": pos.realised_pnl,
290                "total_pnl": pos.total_pnl
291            }
292        return holdings

294    def update_market_value_of_asset(
295        self, asset, current_price, current_dt
296    ):
297        """
298        Update the market value of the asset to the current
299        trade price and date.
300        """
301        if asset not in self.pos_handler.positions:
302            return
303        else:
304            if current_price < 0.0:
305                raise ValueError(
306                    'Current trade price of %s is negative for '
307                    'asset %s. Cannot update position.' % (
308                        current_price, asset
309                    )
310                )
311
312            if current_dt < self.current_dt:
313                raise ValueError(
314                    'Current trade date of %s is earlier than '
315                    'current date %s of asset %s. Cannot update '
316                    'position.' % (
317                        current_dt, self.current_dt, asset
318                    )
319                )
320
321            self.pos_handler.positions[asset].update_current_price(
322                current_price, current_dt
323            )

325    def history_to_df(self):
326        """
327        Creates a Pandas DataFrame of the Portfolio history.

329        records = [pe.to_dict() for pe in self.history]

205    def transact_asset(self, txn):
206        """
207        Adjusts positions to account for a transaction.
208        """
209        if txn.dt < self.current_dt:
210            raise ValueError(
211                'Transaction datetime (%s) is earlier than '
212                'current portfolio datetime (%s). Cannot '
213                'transact assets.' % (txn.dt, self.current_dt)
214            )
215        self.current_dt = txn.dt
216
217        txn_share_cost = txn.price * txn.quantity
218        txn_total_cost = txn_share_cost + txn.commission
219
220        if txn_total_cost > self.cash:
221            if settings.PRINT_EVENTS:
222                print(
223                    'WARNING: Not enough cash in the portfolio to '
224                    'carry out transaction. Transaction cost of %s '
225                    'exceeds remaining cash of %s. Transaction '
226                    'will proceed with a negative cash balance.' % (
227                        txn_total_cost, self.cash
228                    )
229                )
230
231        self.pos_handler.transact_position(txn)
232
233        self.cash -= txn_total_cost
234
235        # Form Portfolio history details
236        direction = "LONG" if txn.direction > 0 else "SHORT"
237        description = "%s %s %s %0.2f %s" % (
238            direction, txn.quantity, txn.asset.upper(),
239            txn.price, datetime.datetime.strftime(txn.dt, "%d/%m/%Y")
240        )
241        if direction == "LONG":
242            pe = PortfolioEvent(
243                dt=txn.dt, type='asset_transaction',
244                description=description,
245                debit=round(txn_total_cost, 2), credit=0.0,
246                balance=round(self.cash, 2)
247            )
248            self.logger.info(
249                '(%s) Asset "%s" transacted LONG in portfolio "%s" '
250                '- Debit: %0.2f, Balance: %0.2f' % (
251                    txn.dt.strftime(settings.LOGGING["DATE_FORMAT"]),
252                    txn.asset, self.portfolio_id,
253                    round(txn_total_cost, 2), round(self.cash, 2)
254                )
255            )
256        else:
257            pe = PortfolioEvent(
258                dt=txn.dt, type='asset_transaction',
259                description=description,
260                debit=0.0, credit=-1.0 * round(txn_total_cost, 2),
261                balance=round(self.cash, 2)
262            )
263            self.logger.info(
264                '(%s) Asset "%s" transacted SHORT in portfolio "%s" '
265                '- Credit: %0.2f, Balance: %0.2f' % (
266                    txn.dt.strftime(settings.LOGGING["DATE_FORMAT"]),
267                    txn.asset, self.portfolio_id,
268                    -1.0 * round(txn_total_cost, 2), round(self.cash, 2)
269                )
270            )
271        self.history.append(pe)
